\hfil Theories of verb raising

\medskip
\\Trace theory

\leftside
\jtree
\! = :{Agr} !a :{$t_i$} :{$t_j$}() \etc.
\psset{scaleby=.3 .4}
\!a = :{Agr} {Tns$_i$} :{Tns} {\Vj}.
\endjtree

\\Copy theory

\leftside
\jtree
\! = :{Agr}!a [scaleby=1.45]
   :{Tns$_i$}!b :{\Vj}() \etc.
\psset{scaleby=.3 .4}
\!a = :{Agr} {Tns$_i$} :{Tns} {\Vj}.
\!b = :{Tns} {\Vj}.
\endjtree

\\Shared structure

\leftside
\jtree
\! = :{Agr}@A1 !a :{Tns}@A2 !b :{V}@A3 \etc.
\psset{scaleby=.3 .4,angleA=-35,angleB=125,ncurv=1.5,nodesep=0}
\!a = <left>{Agr}.
\!b = <left>{Tns}.
\nccurve{->}{A1:b}{A2:t}
\nccurve{->}{A2:b}{A3:t}
\endjtree
